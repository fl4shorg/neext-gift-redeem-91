    <script>
        document.getElementById('redeemForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            const resgatante = document.getElementById('resgatante').value;
            const gift = document.getElementById('gift').value;
            const resultDiv = document.getElementById('result');
            
            try {
                const response = await fetch(`https://script.google.com/macros/s/AKfycbwZYzU01xSMDJSTg7z5-Gyj7c99ffaEAbAJPwjnrPYEI1hce2nAOKJiYOTd3n72qjXG/exec?action=redeem&resgatante=${encodeURIComponent(resgatante)}&gift=${encodeURIComponent(gift)}`);
                
                const data = await response.json();
                
                if (data.status === "success") {
                    let html = "<h3>Resgate realizado com sucesso!</h3>";
                    
                    // Verificar se é conta IPTV (baseado no que o AppScript retorna)
                    if (data.hasOwnProperty('servidor')) {
                        // É conta IPTV - mostra nome, senha e servidor
                        html += `
                            <div class="info-row"><span class="info-label">Tipo:</span> IPTV</div>
                            <div class="info-row"><span class="info-label">Usuário:</span> ${data.nome || 'Não informado'}</div>
                            <div class="info-row"><span class="info-label">Senha:</span> ${data.senha || 'Não informada'}</div>
                            <div class="info-row"><span class="info-label">Servidor:</span> ${data.servidor || 'Não informado'}</div>
                        `;
                    } else {
                        // É conta normal - mostra email e senha
                        html += `
                            <div class="info-row"><span class="info-label">Tipo:</span> ${data.tipo || 'Normal'}</div>
                            <div class="info-row"><span class="info-label">Email:</span> ${data.email || 'Não informado'}</div>
                            <div class="info-row"><span class="info-label">Senha:</span> ${data.senha || 'Não informada'}</div>
                        `;
                    }
                    
                    html += `
                        <div class="info-row"><span class="info-label">GiftCard:</span> ${data.gift || 'Não informado'}</div>
                        <div class="info-row"><span class="info-label">Resgatante:</span> ${data.resgatante || 'Não informado'}</div>
                        <div class="info-row"><span class="info-label">Data/Hora:</span> ${data.dataHora || 'Não informada'}</div>
                    `;
                    
                    resultDiv.innerHTML = html;
                    resultDiv.className = "result success";
                } else {
                    resultDiv.innerHTML = `<p>Erro: ${data.message}</p>`;
                    resultDiv.className = "result error";
                }
            } catch (error) {
                resultDiv.innerHTML = `<p>Erro de conexão: ${error.message}</p>`;
                resultDiv.className = "result error";
            }
            
            resultDiv.style.display = "block";
        });
    </script>
